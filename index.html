import time import threading import tkinter as tk from twilio.rest import Client # Twilio config account_sid = "YOUR_SID" auth_token = "YOUR_TOKEN" client = Client(account_sid, auth_token) numbers = ["+8490xxxxxxx", "+8491xxxxxxx"] paused = False stopped = False def send_sms(): global paused, stopped for num in numbers: if stopped: print("⛔ Đã dừng hoàn toàn") break while paused: time.sleep(1) client.messages.create( body="Thong bao hop le", from_="+1XXXXXXXXXX", to=num ) print("Sent to", num) time.sleep(3) def start(): global stopped stopped = False threading.Thread(target=send_sms).start() def pause(): global paused paused = True print("⏸️ Pause") def resume(): global paused paused = False print("▶️ Resume") def stop(): global stopped stopped = True print("⛔ Stop") # GUI root = tk.Tk() root.title("SMS Sender") tk.Button(root, text="Start", command=start, width=15).pack(pady=5) tk.Button(root, text="Pause", command=pause, width=15).pack(pady=5) tk.Button(root, text="Resume", command=resume, width=15).pack(pady=5) tk.Button(root, text="Stop", command=stop, width=15).pack(pady=5) root.mainloop() có
